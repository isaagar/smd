		package ccoew.it.i3.businesslogic;

		import java.util.ArrayList;

		public class Route {
	
			private int routeNo;
			private BusTerminus startTerminus;
			private BusTerminus stopTerminus;
			private ArrayList<BusStop> busStops;
			public Route(int routeNo, BusTerminus startTerminus,
					BusTerminus stopTerminus, ArrayList<BusStop> busStops) {
				super();
				this.routeNo = routeNo;
				this.startTerminus = startTerminus;
				this.stopTerminus = stopTerminus;
				this.busStops = busStops;
			}
			public int getRouteNo() {
				return routeNo;
			}
			private void setRouteNo(int routeNo) {
				this.routeNo = routeNo;
			}
			public BusTerminus getStartTerminus() {
				return startTerminus;
			}
			private void setStartTerminus(BusTerminus startTerminus) {
				this.startTerminus = startTerminus;
			}
			public BusTerminus getStopTerminus() {
				return stopTerminus;
			}
			private void setStopTerminus(BusTerminus stopTerminus) {
				this.stopTerminus = stopTerminus;
			}
			public ArrayList<BusStop> getBusStops() {
				return busStops;
			}
			private void setBusStops(ArrayList<BusStop> busStops) {
				this.busStops = busStops;
			}
			@Override
			public String toString() {
				StringBuilder builder = new StringBuilder();
				builder.append("Route [routeNo=");
				builder.append(routeNo);
		
				builder.append(", startTerminus=");
				builder.append(startTerminus);
		
				builder.append(", stopTerminus=");
				builder.append(stopTerminus);
		
				builder.append(", busStops=");
				//builder.append(super.toString());
				builder.append(busStops);
				builder.append("]");
				return builder.toString();
			}
	
			public void display(){
				System.out.println(this.toString());
			}
		}



		TransportSystem.java

		package ccoew.it.i3.businesslogic;

		import java.util.ArrayList;

		public class TransportSystem {
	
			private ArrayList<Bus> buses;
			private ArrayList<Route> routes;
			private ArrayList<BusTerminus> terminuses;
			public TransportSystem(ArrayList<Bus> buses, ArrayList<Route> routes,
					ArrayList<BusTerminus> terminuses) {
				super();
				this.buses = buses;
				this.routes = routes;
				this.terminuses = terminuses;
			}
			public ArrayList<Bus> getBuses() {
				return buses;
			}
			private void setBuses(ArrayList<Bus> buses) {
				this.buses = buses;
			}
			public ArrayList<Route> getRoutes() {
				return routes;
			}
			private void setRoutes(ArrayList<Route> routes) {
				this.routes = routes;
			}
			public ArrayList<BusTerminus> getTerminuses() {
				return terminuses;
			}
			private void setTerminuses(ArrayList<BusTerminus> terminuses) {
				this.terminuses = terminuses;
			}
			@Override
			public String toString() {
				StringBuilder builder = new StringBuilder();
				builder.append("TransportSystem [buses=");
				builder.append(buses);
		
				builder.append(", routes=");
				builder.append(routes);
		
				builder.append(", terminuses=");
				builder.append(terminuses);
				builder.append("]");
				return builder.toString();
			}
	
			public void display()
			{
				System.out.println(this.toString());
			}
		}
